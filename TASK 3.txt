import java.util.Scanner;

class Student {

    private int rollNo;
    private String name;
    private double marks;
    private int attendance;

    public Student() {
    }

    public void setRollNo(int r) {
        rollNo = r;
    }

    public void setName(String n) {
        name = n;
    }

    public void setMarks(double m) {
        marks = m;
    }

    public void setAttendance(int a) {
        attendance = a;
    }

    public int getRollNo() {
        return rollNo;
    }

    public String getName() {
        return name;
    }

    public double getMarks() {
        return marks;
    }

    public int getAttendance() {
        return attendance;
    }

    public void calculateResult(int totalDays) {
        if (marks >= 40 && attendance >= (totalDays * 75 / 100)) {
            System.out.println("Result: Pass");
        } else {
            System.out.println("Result: Fail");
        }
    }
}

public class SchoolRecord {

    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);

        int totalDays = 100;

        Student st1 = new Student();
        Student st2 = new Student();
        Student st3 = new Student();

        st1.setRollNo(1);
        st1.setName("Ahmed");
        st1.setMarks(55);
        st1.setAttendance(80);

        st2.setRollNo(2);
        st2.setName("Maria");
        st2.setMarks(35);
        st2.setAttendance(90);

        st3.setRollNo(3);
        st3.setName("David");
        st3.setMarks(75);
        st3.setAttendance(60);

        Student[] list = {st1, st2, st3};

        for (int i = 0; i < list.length; i++) {
            System.out.println("Roll No: " + list[i].getRollNo());
            System.out.println("Name: " + list[i].getName());
            System.out.println("Marks: " + list[i].getMarks());
            System.out.println("Attendance: " + list[i].getAttendance());
            list[i].calculateResult(totalDays);
            System.out.println();
        }

        input.close();
    }
}
