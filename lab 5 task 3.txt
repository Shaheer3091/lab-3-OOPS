class Employee {

    static double bonusPool = 0;
    static double bonusPercentage = 10;
    static int totalEmployees = 0;

    int empId;
    String name;
    double basicSalary;

    Employee(int empId, String name, double basicSalary) {
        this.empId = empId;
        this.name = name;
        this.basicSalary = basicSalary;

        totalEmployees++;
        bonusPool = bonusPool + (basicSalary * bonusPercentage / 100);
    }

    double calculateTotalCompensation() {
        return basicSalary + (bonusPool / totalEmployees);
    }

    static void displayBonusPool() {
        System.out.println("Total Bonus Pool: " + bonusPool);
        System.out.println();
    }
}

public class Main {
    public static void main(String[] args) {

        Employee e1 = new Employee(1, "Ali", 50000);
        Employee e2 = new Employee(2, "Ahmed", 60000);
        Employee e3 = new Employee(3, "Haris", 55000);
        Employee e4 = new Employee(4, "Hina", 70000);

        System.out.println("Employee ID: " + e1.empId);
        System.out.println("Name: " + e1.name);
        System.out.println("Basic Salary: " + e1.basicSalary);
        System.out.println("Total Compensation: " + e1.calculateTotalCompensation());
        System.out.println();

        System.out.println("Employee ID: " + e2.empId);
        System.out.println("Name: " + e2.name);
        System.out.println("Basic Salary: " + e2.basicSalary);
        System.out.println("Total Compensation: " + e2.calculateTotalCompensation());
        System.out.println();

        System.out.println("Employee ID: " + e3.empId);
        System.out.println("Name: " + e3.name);
        System.out.println("Basic Salary: " + e3.basicSalary);
        System.out.println("Total Compensation: " + e3.calculateTotalCompensation());
        System.out.println();

        System.out.println("Employee ID: " + e4.empId);
        System.out.println("Name: " + e4.name);
        System.out.println("Basic Salary: " + e4.basicSalary);
        System.out.println("Total Compensation: " + e4.calculateTotalCompensation());
        System.out.println();

        Employee.displayBonusPool();
    }
}
