class Member {

    private int memberID;
    private String name;
    private double weight;
    private int workoutHours;

    public Member() {
    }

    public void setMemberID(int id) {
        memberID = id;
    }

    public void setName(String n) {
        name = n;
    }

    public void setWeight(double w) {
        weight = w;
    }

    public void setWorkoutHours(int h) {
        workoutHours = h;
    }

    public int getMemberID() {
        return memberID;
    }

    public String getName() {
        return name;
    }

    public double getWeight() {
        return weight;
    }

    public int getWorkoutHours() {
        return workoutHours;
    }

    public void assessProgress(double oldWeight) {
        if (workoutHours >= 10 && weight < oldWeight) {
            System.out.println("Assessment: Good Progress");
        } else {
            System.out.println("Assessment: Needs Improvement");
        }
    }
}

public class FitnessCenter {

    public static void main(String[] args) {

        Member mem1 = new Member();
        Member mem2 = new Member();
        Member mem3 = new Member();

        mem1.setMemberID(1);
        mem1.setName("Usman");
        mem1.setWeight(72);
        mem1.setWorkoutHours(15);

        mem2.setMemberID(2);
        mem2.setName("Hina");
        mem2.setWeight(65);
        mem2.setWorkoutHours(6);

        mem3.setMemberID(3);
        mem3.setName("Bilal");
        mem3.setWeight(80);
        mem3.setWorkoutHours(11);

        double oldWeight1 = 75;
        double oldWeight2 = 66;
        double oldWeight3 = 85;

        Member[] members = {mem1, mem2, mem3};
        double[] oldWeights = {oldWeight1, oldWeight2, oldWeight3};

        for (int i = 0; i < members.length; i++) {

            System.out.println("Member ID: " + members[i].getMemberID());
            System.out.println("Name: " + members[i].getName());
            System.out.println("Current Weight: " + members[i].getWeight());
            System.out.println("Workout Hours: " + members[i].getWorkoutHours());

            members[i].assessProgress(oldWeights[i]);
            System.out.println();
        }
    }
}
